// 增加元素 hashMap.put("name", "老王"); hashMap.put("age", "30"); hashMap.put("sex", "你猜"); // 删除元素 hashMap.remove("age"); // 查找单个元素 System.out.println(hashMap.get("age")); // 循环所有的 key for (Object k : hashMap.keySet()) { System.out.println(k);
